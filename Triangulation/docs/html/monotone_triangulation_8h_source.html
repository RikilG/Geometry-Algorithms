151 vector<Triangle> monotoneTriangulate(vector<Vertex> input) {
152  vector<Triangle> output;
153  Vertex left = findMin(input), right = findMax(input);
154  // assign chain sides and sort input w.r.t x axis
155  input = preprocess(input, left, right);
156  list<Vertex> stk; // queue/stack chain (also the reflex chain)
157  unsigned long i=0;
158  while(i != input.size()) {
159  if (stk.size() < 2) {
160  stk.push_back(input[i]);
161  i++;
162  continue;
163  }
164  Vertex p, u; // p = vertex eliminated if triangulated, u = adjoining to p
165  Vertex v = input[i]; // new vertex
166  bool vSide = v.tag;
167  if (vSide == stk.back().tag) { // same side
168  while (stk.size() >= 2) {
169  p = stk.back();
170  stk.pop_back();
171  u = stk.back();
172  if (validTriangle(u, p, v, p.tag)) {
173  output.emplace_back(u, p, v);
174  } else {
175  stk.push_back(p); // push the popped vertex to chain
176  break; // since we can't form proper triangles
177  }
178  }
179  } else { // opposite side, just add diagonals to triangulate
180  while (stk.size() >= 2) {
181  p = stk.front();
182  stk.pop_front();

184  output.emplace_back(v, p, u);

186  }

188  i++;
